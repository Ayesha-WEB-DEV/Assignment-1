def isPalindrome(string):
    rev = ''.join(reversed(string))
    if string == rev:
        return True
    return False


def prime_factors(n):
    i = 2
    factors = []
    while i * i <= n:
        if n % i:
            i += 1
        else:
            n //= i
            factors.append(i)
    if n > i:
        factors.append(n)
    return factors
